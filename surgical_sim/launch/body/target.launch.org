<launch>
  <arg name="model" default="$(find surgical_sim)/models/urdf/target_object/target_object.urdf.xacro" />
  <arg name="gazebo_gui" default="true" />
  <arg name="id" default="right" />
  <arg name="rviz" default="true"/>
  <arg name="virtual_target" default="true"/>
  <arg name="tracking_mode" default="right"/>
  <arg name="target_name" default="targetx"/>
  <arg name="subjectID" default="s0"/>
  <arg name="spawn_delay" default="0"/>

  
  <arg name="freq" default="300.0"/>  
  
  

  <group ns="$(arg id)_target">
    <param name="robot_description" command="rosrun xacro xacro --inorder $(arg model)"/>
    <node pkg="robot_state_publisher" type="robot_state_publisher"  name="target_object_state_publisher">
      <param name="publish_frequency" type="double" value="$(eval 2*arg('freq'))" />
    </node>
  </group>


  <group if="$(arg virtual_target)">
    <node pkg="tf2_ros" type="static_transform_publisher" name="baseObject_static_tf" args="0 0.5 1.0 0 0 0 world $(arg id)_target_base_link" />
  </group>
  
  <group unless="$(arg virtual_target)">
    <node name="target_state_publisher" pkg="surgical_sim" type="targetObject" output="screen">
      <param name="trackingMode" value="$(arg tracking_mode)"/>
      <param name="targetName" value="$(arg target_name)"/>
      <param name="subjectID" value="$(arg subjectID)"/>
      <param name="publish_frequency" type="double" value="$(arg freq)" />
      <remap from="robot_description" to="target_object_description"/>
      <remap from="joint_states" to="target_state_publisher/joint_states"/>
    </node> 
  </group>
   

  <group if="$(arg rviz)">
    <node name="rviz" pkg="rviz" type="rviz" args="-d $(find surgical_sim)/rviz/target_object_view.rviz" />
  </group>

  <include file="$(find surgical_sim)/launch/robot.gazebo.launch">
      <arg name="gui" value="$(arg gazebo_gui)"/>
      <arg name="w_world" value="$(arg rviz)"/>  
  </include> 

  <node name="spawn_$(arg id)_target" pkg="gazebo_ros" type="spawn_model" launch-prefix="bash -c 'sleep $(arg spawn_delay); $0 $@'" 
        args="-urdf -x 0.0 -y 0.5 -z 1.2 -model $(arg id)_target -param $(arg id)_target/robot_description " respawn="false" output="screen" />

</launch>
